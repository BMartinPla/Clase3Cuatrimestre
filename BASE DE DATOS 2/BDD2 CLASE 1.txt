create table genero (
	id int not null identity (1,1) primary key,
	nombre varchar(32) not null unique
)

create table estCivil (
	id int not null identity (1,1) primary key,
	nombre varchar(32) not null unique
)

create table pais (
	id int not null identity (1,1) primary key,
	nombre varchar(32) not null unique
)

create table tipoDoc (
	id int not null identity (1,1) primary key,
	pais_id integer not null,
	nombre varchar(32) not null unique,

constraint fk_pais foreign key (pais_id) references pais(id)
)

create table tipoCont (
	id int not null identity (1,1) primary key,
	nombre varchar(32) not null unique
)

create table persona (
	id int not null identity (1,1) primary key,
	apellido varchar (32) not null,
	nombre varchar(32) not null,
	tipoDoc_id int not null,
	documento varchar(32) not null,
	genero_id int not null,
	estCivil_id int not null,
	tipoCont_id int not null,
	contacto varchar(64) not null,

constraint fk_tipoDoc_persona foreign key (tipoDoc_id) references tipoDoc(id),
constraint fk_genero_persona foreign key (genero_id) references genero(id),
constraint fk_estCivil_persona foreign key (estCivil_id) references estCivil(id)
)

alter table persona
add constraint uq_documento unique (tipoDoc_id, documento)

create table contacto (
	id int not null identity (1,1) primary key,
	persona_id int not null,
	tipoCont_id int not null,
	contacto varchar(64) not null,
	
constraint fk_contacto_persona foreign key (persona_id) references persona(id),
constraint fk_tipoCont_persona foreign key (tipoCont_id) references tipoCont(id)
)

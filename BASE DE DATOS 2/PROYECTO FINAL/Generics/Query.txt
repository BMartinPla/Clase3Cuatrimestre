-- Vistas
create view vPais as
select id as paisid, nombre
from Pais;

select * from vPais

-- Vistas complejas
create view vTipoDocumento as
select
	td.id as tipodocumentoid,
	td.nombre as tipodocumento,
	p.id as pais_id,
	p.nombre as nombrepais
from TipoDocumento td
inner join Pais p
	on td.pais_id = p.id;

select * from vTipoDocumento

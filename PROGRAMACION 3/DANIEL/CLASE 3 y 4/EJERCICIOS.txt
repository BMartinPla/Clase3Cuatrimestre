-- ej1: Reporte Alumnos ordenados por Apellido y Nombre

SELECT * from alumnos a order by apellido asc, nombre asc

-- ej2: Reporte Alumnos ordenados por Apellido y Nombre, cuyo apellido comience con XX

SELECT * from alumnos a where apellido like 'T%' order by apellido asc, nombre asc 

-- ej3: Reporte Alumnos ordenados por Apellido y Nombre, cuyo apellido comience con la letra A,
-- Reporte: Apellido y Nombre-DNI-direccion

SELECT apellido, nombre, dni, CONCAT (a.calle, ' ', a.nro, ' ', a.unidad) as direccion from alumnos a where apellido like 'A%' order by apellido asc, nombre asc 

-- ej4: Reporte Alumnos ordenados por Apellido y Nombre, con domiclio en La Emilia 2901,
-- Reporte: Apellido y Nombre-DNI-direccion-Localidad

SELECT a.apellido, a.nombre, a.dni, a.cod_postal, CONCAT (a.calle, ' ', a.nro, ' ', a.unidad) as direccion
from alumnos a
where a.cod_postal like '2901' order by apellido asc, nombre asc 

-- ej5: Reporte Alumnos ordenados descendente por edad. Utilice una funcion para el calculo de la edad.
-- Reporte: Apellido y Nombre-DNI-edad-direccion-Localidad

SELECT a.apellido, a.nombre, a.dni, dbo.CalcularEdad(fecha_nacimiento) as edad, CONCAT (a.calle, ' ', a.nro, ' ', a.unidad) as direccion
from alumnos a
order by dbo.CalcularEdad(fecha_nacimiento) desc

-- ej6: Reporte Alumnos con estado Activo por Localidad, ordenado por cantidad descendente.
-- Muestre solo las localidades con mas de 10 alumnos.
-- Reporte: Localidad-Cantidad Alumnos

SELECT a.cod_postal, l.descripcion, count(*) as cant_alumnos 
from alumnos a
inner join localidades l
on a.cod_postal = l.cod_postal
WHERE a.estado LIKE 'ACT'
group by a.cod_postal, l.descripcion
order by cant_alumnos desc

-- ej7: Reporte cantidad Alumnos por rango etario:
-- Rangos: (<20) - ( > 20 <= 25 ) - ( > 25 <= 30 ) - ( > 30 <= 35 ) - ( > 35 <= 40 ) - ( > 40 )
-- Reporte: Rango-Cantidad Alumnos

SELECT dbo.CalcularRangoEdad(fecha_nacimiento) as rango, count(*)
from alumnos a
group by dbo.CalcularRangoEdad(fecha_nacimiento)

-- ej8: Alumnos por comision
-- Reporte: curso - anio - turno - comision - legajo - apynom - telefono

SELECT c.curso, c.anio, c.turno, c.comision, a.nro_legajo, CONCAT (RTRIM(a.apellido), ' ', RTRIM(a.nombre)), a.telefono
from alumnos a
inner join cursos_alumnos ca
on a.nro_alumno = ca.nro_alumno
inner join cursos c
on ca.curso = c.curso
order by c.comision

-- ej9: Materias por curso
-- Reporte: curso - anio - turno - Materia 

select c.curso, c.anio, c.turno, m.desc_corta
from materias m
inner join cursos_materias cm
on m.materia = cm.materia
inner join cursos c
on cm.curso = c.curso
order by c.anio asc, m.desc_corta asc

-- ej10: Cantidad alumnos comisiones
-- Reporte: curso - anio - turno - cantidad alumnos 

select c.curso, c.anio, c.turno, count(a.nro_alumno) as cantidad
from alumnos a
inner join cursos_alumnos ca
on a.nro_alumno = ca.nro_alumno
inner join cursos c
on ca.curso = c.curso
group by c.curso, c.anio, c.turno

-- ej11: Promedio notas alumnos ordenados de mayor a menor
-- Reporte: nro. alumno - apellido y nombre - promedio

select a.nro_alumno, CONCAT (RTRIM(a.apellido), ' ', RTRIM(a.nombre)) as apynom, avg(ae.nota) as promedio
from alumnos a
inner join alumnos_examenes ae
on a.nro_alumno = ae.nro_alumno
group by a.nro_alumno, a.apellido, a.nombre
order by promedio desc

-- ej12: Promedio notas alumnos ordenados de mayor a menor solo los mayores de 7.
-- Reporte: nro. alumno - apellido y nombre - promedio


select a.nro_alumno, CONCAT (RTRIM(a.apellido), ' ', RTRIM(a.nombre)) as apynom, avg(ae.nota) as promedio
from alumnos a
inner join alumnos_examenes ae
on a.nro_alumno = ae.nro_alumno
group by a.nro_alumno, a.apellido, a.nombre
having avg(ae.nota) > 7
order by promedio desc
